import random

class JankenGame:
    """じゃんけんゲームクラス"""
    
    HANDS = {
        1: "グー",
        2: "チョキ", 
        3: "パー"
    }
    
    def __init__(self):
        self.player_wins = 0
        self.computer_wins = 0
        self.draws = 0
    
    def get_computer_hand(self):
        """コンピュータの手をランダムに生成"""
        return random.randint(1, 3)
    
    def judge_winner(self, player_hand, computer_hand):
        """勝敗判定
        
        Args:
            player_hand (int): プレイヤーの手 (1:グー, 2:チョキ, 3:パー)
            computer_hand (int): コンピュータの手 (1:グー, 2:チョキ, 3:パー)
            
        Returns:
            str: 判定結果 ("player_win", "computer_win", "draw")
        """
        if player_hand == computer_hand:
            return "draw"
        
        # プレイヤーが勝つパターン
        win_patterns = {
            (1, 2),  # グー vs チョキ
            (2, 3),  # チョキ vs パー
            (3, 1)   # パー vs グー
        }
        
        if (player_hand, computer_hand) in win_patterns:
            return "player_win"
        else:
            return "computer_win"
    
    def is_valid_input(self, user_input):
        """入力値の妥当性チェック
        
        Args:
            user_input (str): ユーザー入力
            
        Returns:
            bool: 有効な入力の場合True
        """
        return user_input in ["1", "2", "3", "q"]
    
    def play_round(self, player_hand):
        """1ラウンドのゲーム実行
        
        Args:
            player_hand (int): プレイヤーの手
            
        Returns:
            dict: ゲーム結果の辞書
        """
        computer_hand = self.get_computer_hand()
        result = self.judge_winner(player_hand, computer_hand)
        
        # 勝敗カウント更新
        if result == "player_win":
            self.player_wins += 1
        elif result == "computer_win":
            self.computer_wins += 1
        else:
            self.draws += 1
        
        return {
            "player_hand": player_hand,
            "computer_hand": computer_hand,
            "result": result
        }
    
    def display_result(self, game_result):
        """ゲーム結果を表示"""
        player_hand_name = self.HANDS[game_result["player_hand"]]
        computer_hand_name = self.HANDS[game_result["computer_hand"]]
        
        print(f"あなた: {player_hand_name}")
        print(f"コンピュータ: {computer_hand_name}")
        
        if game_result["result"] == "player_win":
            print("あなたの勝ちです！")
        elif game_result["result"] == "computer_win":
            print("コンピュータの勝ちです！")
        else:
            print("引き分けです！")
    
    def display_score(self):
        """現在のスコアを表示"""
        print(f"\n現在のスコア - あなた: {self.player_wins}勝, コンピュータ: {self.computer_wins}勝, 引き分け: {self.draws}回")
    
    def run(self):
        """メインゲームループ"""
        print("じゃんけんゲームへようこそ！")
        print("1: グー, 2: チョキ, 3: パー, q: 終了")
        
        while True:
            print("\n" + "="*30)
            user_input = input("あなたの手を選んでください: ").strip()
            
            if not self.is_valid_input(user_input):
                print("無効な入力です。1, 2, 3, またはqを入力してください。")
                continue
            
            if user_input == "q":
                print("\nゲーム終了！")
                self.display_score()
                print("ありがとうございました！")
                break
            
            player_hand = int(user_input)
            game_result = self.play_round(player_hand)
            
            print()
            self.display_result(game_result)
            self.display_score()

def main():
    """メイン関数"""
    game = JankenGame()
    game.run()

if __name__ == "__main__":
    main()

import unittest
from unittest.mock import patch
from janken import JankenGame

class TestJankenGame(unittest.TestCase):
    """じゃんけんゲームのテストクラス"""
    
    def setUp(self):
        """各テストメソッド実行前の初期化"""
        self.game = JankenGame()
    
    def test_judge_winner_all_patterns(self):
        """勝敗判定ロジックの全パターンテスト（9通り）"""
        
        # 引き分けパターン（3通り）
        self.assertEqual(self.game.judge_winner(1, 1), "draw")  # グー vs グー
        self.assertEqual(self.game.judge_winner(2, 2), "draw")  # チョキ vs チョキ
        self.assertEqual(self.game.judge_winner(3, 3), "draw")  # パー vs パー
        
        # プレイヤー勝利パターン（3通り）
        self.assertEqual(self.game.judge_winner(1, 2), "player_win")  # グー vs チョキ
        self.assertEqual(self.game.judge_winner(2, 3), "player_win")  # チョキ vs パー
        self.assertEqual(self.game.judge_winner(3, 1), "player_win")  # パー vs グー
        
        # コンピュータ勝利パターン（3通り）
        self.assertEqual(self.game.judge_winner(2, 1), "computer_win")  # チョキ vs グー
        self.assertEqual(self.game.judge_winner(3, 2), "computer_win")  # パー vs チョキ
        self.assertEqual(self.game.judge_winner(1, 3), "computer_win")  # グー vs パー
    
    def test_is_valid_input(self):
        """入力値検証のテスト"""
        
        # 有効な入力
        self.assertTrue(self.game.is_valid_input("1"))
        self.assertTrue(self.game.is_valid_input("2"))
        self.assertTrue(self.game.is_valid_input("3"))
        self.assertTrue(self.game.is_valid_input("q"))
        
        # 無効な入力
        self.assertFalse(self.game.is_valid_input("0"))
        self.assertFalse(self.game.is_valid_input("4"))
        self.assertFalse(self.game.is_valid_input("a"))
        self.assertFalse(self.game.is_valid_input("Q"))  # 大文字
        self.assertFalse(self.game.is_valid_input(""))   # 空文字
        self.assertFalse(self.game.is_valid_input(" "))  # スペース
        self.assertFalse(self.game.is_valid_input("11")) # 複数文字
    
    def test_get_computer_hand_range(self):
        """コンピュータの手生成が1-3の範囲内であることの確認"""
        
        # 100回実行して全て1-3の範囲内であることを確認
        for _ in range(100):
            computer_hand = self.game.get_computer_hand()
            self.assertIn(computer_hand, [1, 2, 3])
            self.assertGreaterEqual(computer_hand, 1)
            self.assertLessEqual(computer_hand, 3)
    
    def test_get_computer_hand_distribution(self):
        """コンピュータの手生成の分布テスト（統計的確認）"""
        
        # 1000回実行して各手が一定回数出現することを確認
        hands_count = {1: 0, 2: 0, 3: 0}
        
        for _ in range(1000):
            hand = self.game.get_computer_hand()
            hands_count[hand] += 1
        
        # 各手が最低200回は出現することを確認（統計的妥当性）
        for count in hands_count.values():
            self.assertGreaterEqual(count, 200)  # 1000回中200回以上（期待値333回の約60%以上）
    
    @patch('janken.JankenGame.get_computer_hand')
    def test_play_round_player_win(self, mock_computer_hand):
        """1ラウンドテスト - プレイヤー勝利"""
        
        # コンピュータがチョキを出すようにモック設定
        mock_computer_hand.return_value = 2
        
        result = self.game.play_round(1)  # プレイヤーはグー
        
        self.assertEqual(result["player_hand"], 1)
        self.assertEqual(result["computer_hand"], 2)
        self.assertEqual(result["result"], "player_win")
        self.assertEqual(self.game.player_wins, 1)
        self.assertEqual(self.game.computer_wins, 0)
        self.assertEqual(self.game.draws, 0)
    
    @patch('janken.JankenGame.get_computer_hand')
    def test_play_round_computer_win(self, mock_computer_hand):

        # コンピュータがパーを出すようにモック設定
