--- calculator.py ---
import re

class Calculator:
    def __init__(self):
        self.pattern = r'^(-?\d+(?:\.\d+)?)\s*([\+\-\*/])\s*(-?\d+(?:\.\d+)?)$'
    
    def parse_expression(self, expression):
        """数式を解析して数値と演算子を抽出"""
        expression = expression.strip()
        match = re.match(self.pattern, expression)
        
        if not match:
            raise ValueError("不正な数式です")
        
        num1 = float(match.group(1))
        operator = match.group(2)
        num2 = float(match.group(3))
        
        return num1, operator, num2
    
    def calculate(self, expression):
        """数式を計算して結果を返す"""
        try:
            num1, operator, num2 = self.parse_expression(expression)
            
            if operator == '+':
                result = num1 + num2
            elif operator == '-':
                result = num1 - num2
            elif operator == '*':
                result = num1 * num2
            elif operator == '/':
                if num2 == 0:
                    raise ZeroDivisionError("ゼロで割ることはできません")
                result = num1 / num2
            else:
                raise ValueError("サポートされていない演算子です")
            
            # 整数の場合は整数で表示、小数の場合は小数で表示
            if result == int(result):
                return int(result)
            else:
                return round(result, 10)  # 浮動小数点の誤差を制限
                
        except ValueError as e:
            raise ValueError(f"入力エラー: {e}")
        except ZeroDivisionError as e:
            raise ZeroDivisionError(f"計算エラー: {e}")

def main():
    """メイン電卓ループ"""
    calc = Calculator()
    print("=== 電卓アプリ ===")
    print("使用方法: 数式を入力してください (例: 1+2, 10/3)")
    print("終了するには 'quit' または 'exit' を入力してください")
    print()
    
    while True:
        try:
            expression = input("計算式を入力 > ").strip()
            
            # 終了条件
            if expression.lower() in ['quit', 'exit', 'q']:
                print("電卓を終了します。")
                break
            
            # 空入力をスキップ
            if not expression:
                continue
            
            # 計算実行
            result = calc.calculate(expression)
            print(f"結果: {result}")
            
        except (ValueError, ZeroDivisionError) as e:
            print(f"エラー: {e}")
        except KeyboardInterrupt:
            print("\n電卓を終了します。")
            break
        except Exception as e:
            print(f"予期しないエラー: {e}")
        
        print()  # 改行で見やすく

if __name__ == "__main__":
    main()

import unittest
from calculator import Calculator

class TestCalculator(unittest.TestCase):
    def setUp(self):
        """各テスト前にCalculatorインスタンスを作成"""
        self.calc = Calculator()
    
    def test_addition(self):
        """加算のテスト"""
        self.assertEqual(self.calc.calculate("1+2"), 3)
        self.assertEqual(self.calc.calculate("10+5"), 15)
        self.assertEqual(self.calc.calculate("-5+10"), 5)
        self.assertEqual(self.calc.calculate("0+0"), 0)
        print("[OK] Addition tests passed")
    
    def test_subtraction(self):
        """減算のテスト"""
        self.assertEqual(self.calc.calculate("5-3"), 2)
        self.assertEqual(self.calc.calculate("10-15"), -5)
        self.assertEqual(self.calc.calculate("-5-3"), -8)
        self.assertEqual(self.calc.calculate("0-5"), -5)
        print("[OK] Subtraction tests passed")
    
    def test_multiplication(self):
        """乗算のテスト"""
        self.assertEqual(self.calc.calculate("3*4"), 12)
        self.assertEqual(self.calc.calculate("-2*5"), -10)
        self.assertEqual(self.calc.calculate("0*10"), 0)
        self.assertEqual(self.calc.calculate("-3*-4"), 12)
        print("[OK] Multiplication tests passed")
    
    def test_division(self):
        """除算のテスト"""
        self.assertEqual(self.calc.calculate("10/2"), 5)
        self.assertEqual(self.calc.calculate("15/3"), 5)
        self.assertEqual(self.calc.calculate("-10/2"), -5)
        self.assertEqual(self.calc.calculate("7/2"), 3.5)
        print("[OK] Division tests passed")
    
    def test_decimal_calculations(self):
        """小数点計算のテスト"""
        self.assertEqual(self.calc.calculate("1.5+2.5"), 4)
        self.assertEqual(self.calc.calculate("3.14*2"), 6.28)
        self.assertEqual(self.calc.calculate("10.5/3"), 3.5)
        self.assertEqual(self.calc.calculate("2.7-1.2"), 1.5)
        print("[OK] Decimal calculation tests passed")
    
    def test_zero_division_error(self):
        """ゼロ除算エラーのテスト"""
        with self.assertRaises(ZeroDivisionError):
            self.calc.calculate("5/0")
        
        with self.assertRaises(ZeroDivisionError):
            self.calc.calculate("0/0")
        print("[OK] Zero division error tests passed")
    
    def test_invalid_expressions(self):
        """不正な数式のテスト"""
        # 不正な形式
        with self.assertRaises(ValueError):
            self.calc.calculate("1++2")
        
        with self.assertRaises(ValueError):
            self.calc.calculate("abc+2")
        
        with self.assertRaises(ValueError):
            self.calc.calculate("1+")
        
        with self.assertRaises(ValueError):
            self.calc.calculate("+2")
        
        with self.assertRaises(ValueError):
            self.calc.calculate("1 2 + 3")
        
        with self.assertRaises(ValueError):
            self.calc.calculate("")
        print("[OK] Invalid expression tests passed")
    
    def test_whitespace_handling(self):
        """空白文字の処理テスト"""
        self.assertEqual(self.calc.calculate(" 1 + 2 "), 3)
        self.assertEqual(self.calc.calculate("  10  /  2  "), 5)
        self.assertEqual(self.calc.calculate("\t5\t*\t3\t"), 15)
        print("[OK] Whitespace handling tests passed")
    
    def test_parse_expression(self):
        """数式解析のテスト"""
        num1, op, num2 = self.calc.parse_expression("5+3")
        self.assertEqual(num1, 5)
        self.assertEqual(op, "+")
        self.assertEqual(num2, 3)
        
        num1, op, num2 = self.calc.parse_expression("-2.5 * 4")
        self.assertEqual(num1, -2.5)
        self.assertEqual(op, "*")
        self.assertEqual(num2, 4)
        print("[OK] Parse expression tests passed")
    
    def test_large_numbers(self):
        """大きな数値のテスト"""
        self.assertEqual(self.calc.calculate("1000000+2000000"), 3000000)
        self.assertEqual(self.calc.calculate("999*999"), 998001)
        print("[OK] Large number tests passed")
    
    def test_precision(self):
        """精度のテスト"""
        result = self.calc.calculate("1/3")
        self.assertAlmostEqual(result, 0.3333333333, places=10)
        
        result = self.calc.calculate("2/3")
        self.assertAlmostEqual(result, 0.6666666667, places=10)
        print("[OK] Precision tests passed")

def run_manual_tests():
    """手動でいくつかの計算をテスト"""
    print("\n=== Manual Test Results ===")
    calc = Calculator()
    
    test_cases = [
        ("1+2", 3),
        ("10/3", 3.3333333333),
        ("5*4", 20),
        ("15-7", 8),
        ("1.5+2.5", 4),
        ("3.14*2", 6.28)
    ]
    
    for expression, expected in test_cases:
        try:
            result = calc.calculate(expression)
            if isinstance(expected, float):
                success = abs(result - expected) < 0.0000000001
            else:
                success = result == expected
            
            status = "[OK]" if success else "[FAIL]"
            print(f"{status} {expression} = {result} (expected: {expected})")
        except Exception as e:
            print(f"[FAIL] {expression} raised {type(e).__name__}: {e}")
    
    # エラーケースのテスト
    error_cases = ["5/0", "abc+2", "1++2"]
    for expression in error_cases:
        try:
            result = calc.calculate(expression)
            print(f"[FAIL] {expression} should have raised error but got: {result}")
        except Exception as e:
            print(f"[OK] {expression} correctly raised {type(e).__name__}: {e}")

if __name__ == "__main__":
    # unittestの実行
    unittest.main(verbosity=2, exit=False)
    
    # 手動テストの実行
    run_manual_tests()
