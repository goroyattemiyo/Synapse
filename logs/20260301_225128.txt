class SandboxManager:
    """Dockerサンドボックス環境の管理（簡易版）"""
    
    def __init__(self):
        self.workspace = Path(tempfile.mkdtemp())
        logger.info(f"Workspace created at: {self.workspace}")
    
    def write_file(self, path: str, content: str) -> Dict[str, Any]:
        """ファイル書き込み"""
        try:
            file_path = self.workspace / path
            file_path.parent.mkdir(parents=True, exist_ok=True)
            file_path.write_text(content, encoding='utf-8')
            return {"success": True, "message": f"File written: {path}"}
        except Exception as e:
            return {"success": False, "error": str(e)}
    
    def read_file(self, path: str) -> Dict[str, Any]:
        """ファイル読み取り"""
        try:
            file_path = self.workspace / path
            if not file_path.exists():
                return {"success": False, "error": f"File not found: {path}"}
            
            content = file_path.read_text(encoding='utf-8')
            # 100KB制限
            if len(content) > 100000:
                content = content[:100000] + "\n... (truncated)"
            
            return {"success": True, "content": content}
        except Exception as e:
            return {"success": False, "error": str(e)}
    
    def run_command(self, command: str) -> Dict[str, Any]:
        """コマンド実行（60秒タイムアウト）"""
        try:
            result = subprocess.run(
                command,
                shell=True,
                cwd=self.workspace,
                capture_output=True,
                text=True,
                timeout=60
            )
            return {
                "success": True,
                "stdout": result.stdout,
                "stderr": result.stderr,
                "return_code": result.returncode
            }
        except subprocess.TimeoutExpired:
            return {"success": False, "error": "Command timed out (60s)"}
        except Exception as e:
            return {"success": False, "error": str(e)}
    
    def get_files(self) -> Dict[str, str]:
        """全ファイルの内容を取得"""
        files = {}
        for file_path in self.workspace.rglob("*"):
            if file_path.is_file():
                try:
                    relative_path = file_path.relative_to(self.workspace)
                    files[str(relative_path)] = file_path.read_text(encoding='utf-8')
                except Exception as e:
                    files[str(relative_path)] = f"Error reading file: {e}"
        return files
    
    def cleanup(self):
        """ワークスペースのクリーンアップ"""
        try:
            import shutil
            shutil.rmtree(self.workspace)
        except Exception as e:
            logger.warning(f"Failed to cleanup workspace: {e}")

class BaseAgent:
    """基底エージェントクラス"""
    
    def __init__(self, name: str, model: str, api_key: str):
        self.name = name
        self.model = model
        self.client = anthropic.Anthropic(api_key=api_key)
        self.system_prompt = self._get_system_prompt()
    
    def _get_system_prompt(self) -> str:
        raise NotImplementedError
    
    async def process(self, messages: List[Message], context: Dict[str, Any]) -> Message:
        """メッセージを処理して応答を生成"""
        conversation = self._build_conversation(messages, context)
        
        try:
            response = await asyncio.to_thread(
                self.client.messages.create,
                model=self.model,
                max_tokens=4000,
                temperature=0.1,
                system=self.system_prompt,
                messages=conversation
            )
            
            content = response.content[0].text if response.content else ""
            return Message(
                agent=self.name,
                content=content,
                timestamp=time.time(),
                message_type="response"
            )
        except Exception as e:
            logger.error(f"API error in {self.name}: {e}")
            return Message(
                agent=self.name,
                content=f"Error: {str(e)}",
                timestamp=time.time(),
                message_type="error"
            )
    
    def _build_conversation(self, messages: List[Message], context: Dict[str, Any]) -> List[Dict]:
        """会話履歴を構築"""
        conversation = []
        
        # 初回はコンテキストを追加
        if context.get('is_first_turn', False):
            user_goal = context.get('user_goal', '')
            conversation.append({
                "role": "user",
                "content": f"<user_request>{user_goal}</user_request>"
            })
        
        # メッセージ履歴を追加
        for msg in messages[-10:]:  # 最新10件のみ
            if msg.agent == self.name:
                continue  # 自分のメッセージはスキップ
            
            role = "user" if msg.agent == "Orchestrator" else "assistant"
            conversation.append({
                "role": role,
                "content": msg.content
            })
        
        return conversation

class Coder(BaseAgent):
    """実装者エージェント"""
    
    def __init__(self, api_key: str, sandbox: SandboxManager):
        super().__init__("Coder", "claude-3-sonnet-20240229", api_key)
        self.sandbox = sandbox
        self.tools = ["write_file", "read_file", "run_command"]
    
    def _get_system_prompt(self) -> str:
        return """# Identity
You are the Coder, responsible for implementing code based on Orchestrator's instructions.

# Rules
1. Write complete, working Python code with proper error handling
2. Always include comprehensive pytest tests
3. Use write_file to create all necessary files
4. Run tests using run_command to verify functionality
5. Report your progress clearly after each tool use

# Tools Available
- write_file: Create or overwrite files
- read_file: Read existing files
- run_command: Execute shell commands

# Output Format
**ACTION**: [WRITING/TESTING/REPORTING]
**FILES**: [List of files you're working with]
**STATUS**: [Your current status]

Then use tools as needed and provide a summary.
"""
    
    async def process(self, messages: List[Message], context: Dict[str, Any]) -> Message:
        """ツール実行可能な処理"""
        # 通常の処理
        response = await super().process(messages, context)
        
        # ツール実行の判定と実行
        if self._should_execute_tools(response.content):
            tool_calls, tool_results = await self._execute_tools(response.content)
            response.tool_calls = tool_calls
            response.tool_results = tool_results
            
            # ツール結果を含む最終応答を生成
            updated_content = response.content + "\n\n" + self._format_tool_results(tool_results)
            response.content = updated_content
        
        return response
    
    def _should_execute_tools(self, content: str) -> bool:
        """ツール実行が必要かどうか判定"""
        tool_indicators = ["write_file", "read_file", "run_command", "create file", "test"]
        return any(indicator in content.lower() for indicator in tool_indicators)
    
    async def _execute_tools(self, content: str) -> tuple[List[Dict], List[Dict]]:
        """ツールを実行"""
        tool_calls = []
        tool_results = []
        
        # 簡易的なツール抽出ロジック
        lines = content.split('\n')
        for line in lines:
            line = line.strip()
            
            # write_file の検出
            if line.startswith('write_file') or 'create file' in line.lower():
                # ファイル名とコンテンツを抽出（簡易版）
                if 'main.py' in line or 'calculator.py' in line:
                    file_content = self._generate_calculator_code()
                    call = {"tool": "write_file", "path": "calculator.py", "content": file_content}
                    result = self.sandbox.write_file("calculator.py", file_content)
                    
                    tool_calls.append(call)
                    tool_results.append(result)
                
                elif 'test' in line.lower():
                    test_content = self._generate_test_code()
                    call = {"tool": "write_file", "path": "test_calculator.py", "content": test_content}
                    result = self.sandbox.write_file("test_calculator.py", test_content)
                    
                    tool_calls.append(call)
                    tool_results.append(result)
            
            # run_command の検出
            elif 'run' in line.lower() and 'test' in line.lower():
                call = {"tool": "run_command", "command": "python -m pytest test_calculator.py -v"}
                result = self.sandbox.run_command("python -m pytest test_calculator.py -v")
                
                tool_calls.append(call)
                tool_results.append(result)
        
        return tool_calls, tool_results
    
    def _generate_calculator_code(self) -> str:
        """計算機のサンプルコード生成"""
        return '''class Calculator:

        """ツール実行可能な処理"""
        # 通常の処理
        response = await super().process(messages, context)
        
        # ツール実行の判定と実行
        if self._should_execute_tools(response.content):
            tool_calls, tool_results = await self._execute_tools(response.content)
            response.tool_calls = tool_calls
            response.tool_results = tool_results
            
            # ツール結果を含む最終応答を生成
            updated_content = response.content + "\n\n" + self._format_tool_results(tool_results)
            response.content = updated_content
        
        return response
    
    def _should_execute_tools(self, content: str) -> bool:
        """ツール実行が必要かどうか判定"""
        tool_indicators = ["write_file", "read_file", "run_command", "create file", "test"]
        return any(indicator in content.lower() for indicator in tool_indicators)
    
    async def _execute_tools(self, content: str) -> tuple[List[Dict], List[Dict]]:
        """ツールを実行"""
        tool_calls = []
        tool_results = []
        
        # 簡易的なツール抽出ロジック
        lines = content.split('\n')
        for line in lines:
            line = line.strip()
            
            # write_file の検出
            if line.startswith('write_file') or 'create file' in line.lower():
                # ファイル名とコンテンツを抽出（簡易版）
                if 'main.py' in line or 'calculator.py' in line:
                    file_content = self._generate_calculator_code()
                    call = {"tool": "write_file", "path": "calculator.py", "content": file_content}
                    result = self.sandbox.write_file("calculator.py", file_content)
                    
                    tool_calls.append(call)
                    tool_results.append(result)
                
                elif 'test' in line.lower():
                    test_content = self._generate_test_code()
                    call = {"tool": "write_file", "path": "test_calculator.py", "content": test_content}
                    result = self.sandbox.write_file("test_calculator.py", test_content)
                    
                    tool_calls.append(call)
                    tool_results.append(result)
            
            # run_command の検出
            elif 'run' in line.lower() and 'test' in line.lower():
                call = {"tool": "run_command", "command": "python -m pytest test_calculator.py -v"}
                result = self.sandbox.run_command("python -m pytest test_calculator.py -v")
                
                tool_calls.append(call)
                tool_results.append(result)
        
        return tool_calls, tool_results

    def _format_tool_results(self, results: List[Dict]) -> str:
        """ツール結果をフォーマット"""
        output = "\n**TOOL RESULTS:**\n"
        for i, result in enumerate(results, 1):
            if result.get("success"):
                if "stdout" in result:
                    output += f"{i}. Command executed successfully:\n{result['stdout']}\n"
                else:
                    output += f"{i}. {result.get('message', 'Success')}\n"
            else:
                output += f"{i}. Error: {result.get('error')}\n"
        return output
